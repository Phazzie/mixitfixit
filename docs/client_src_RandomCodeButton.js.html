/**
 * @module App
 * @description Main application component.
 */

import React, { useState, useEffect } from "react";
import "./App.css";
import RandomCodeButton from "./components/RandomCodeButton";
import CodeDisplay from "./components/CodeDisplay";
import StatementDisplay from "./components/StatementDisplay";
import { getStatements, saveToLocalStorage } from "./data/dataManager";
import { useStatements } from "./hooks/useStatements";
import { useCode } from "./hooks/useCode";
import { getCode } from "./data/dataManager";

/**
 * @function App
 * @description The main component of the application.
 * @returns {JSX.Element} The rendered App component.
 */
function App() {
  const [statements] = useState(getStatements());
  const { addStatement, removeStatement } = useStatements();
  const { addCode, removeCode } = useCode();
  const [generatedCode, setGeneratedCode] = useState(null);

  /**
   * @function handleGenerateCode
   * @description Handles the code generation process.
   */
  const handleGenerateCode = () => {
    const newCode = getCode();
    setGeneratedCode(newCode);
    addCode(newCode);
  };

  /**
   * @function handleAddStatement
   * @description Handles adding a statement to the statements list.
   * @param {string} newStatement - The new statement to add.
   */
  const handleAddStatement = (newStatement) => {
    addStatement(newStatement);
  };

  /**
   * @function handleRemoveStatement
   * @description Handles removing a statement from the statements list.
   * @param {string} statementToRemove - The statement to remove.
   */
  const handleRemoveStatement = (statementToRemove) => {
    removeStatement(statementToRemove);
  };

  /**
   * @function handleRemoveCode
   * @description Handles removing a code from the codes list.
   * @param {string} codeToRemove - The code to remove.
   */
  const handleRemoveCode = (codeToRemove) => {
    removeCode(codeToRemove);
  };

  useEffect(() => {
    saveToLocalStorage("statements", statements);
  }, [statements]);

  return (
    <div className="App">
      <h1>Mix-It-Fix-It</h1>

      <RandomCodeButton onGenerateCode={handleGenerateCode} />

      <CodeDisplay
        generatedCode={generatedCode}
        onRemoveCode={handleRemoveCode}
      />

      <StatementDisplay
        statements={statements}
        onAddStatement={handleAddStatement}
        onRemoveStatement={handleRemoveStatement}
      />
    </div>
  );
}

export default App;
