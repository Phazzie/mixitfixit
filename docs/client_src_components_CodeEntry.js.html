import React from 'react';

/**
 * CodeEntry Component
 * @module CodeEntry
 * @description Provides a text input for the user to enter a code and a button to submit it.
 */
/**
 * Renders a text input field and a button for code entry.
 *
 * @param {object} props - The component's props.
 * @param {string} props.enteredCode - The current entered code.
 * @param {function} props.setEnteredCode - Function to update the entered code.
 * @param {function} props.setCode - Function to submit the entered code.
 * @param {object} props.rest - Other props to be passed to the button
 * @returns {JSX.Element} The CodeEntry component.
 * @precondition enteredCode is a string.
 * @precondition setEnteredCode is a function.
 * @precondition setCode is a function.
 * @postcondition The component will render a text input and a button.
 */
function CodeEntry({ enteredCode, setEnteredCode, setCode, ...rest }) {
  return (
    <div>
      {/* Enter a code to load a session */}
      <input
        type="text"
        value={enteredCode}
        onChange={(e) => setEnteredCode(e.target.value)} 
        placeholder="Enter code"
        />
        <button onClick={() => setCode(enteredCode)} {...rest}>Use code</button>
    </div>
  );
}

export default CodeEntry;
