import React from 'react';

/**
 * @module StatementList
 * @description This component displays a list of statements.
 * @param {object} props
 * @param {Array<{id: number, user: string, text: string}>} props.statements - An array of statement objects, each with a 'id', 'user' and 'text' property.
 * @returns {JSX.Element} The StatementList component.
 * @preconditions statements must be an array or null
 * @postconditions a list of statements will be rendered.
 */
function StatementList({ statements }) {
  if (!Array.isArray(statements)) {
    console.error("Statements prop is not an array:", statements);
    return <div>Error: Invalid statements data.</div>;
  }

  if (!statements || statements.length === 0) {
    return <div>No statements available.</div>;
  }

  /**
   * Maps over the statements array to create a list item for each statement.
   * @param {object} statement - The statement object
   * @param {number} index - The statement index.
   * @returns {JSX.Element} The statement list element.
   * @preconditions statement must be an object.
   * @postconditions A list item will be created for each statement.
   */
  return (
    <ul>
      {statements.map((statement) => (
        <li key={statement.id}>{`User ${statement.user}: ${statement.text}`}</li>
      ))}
    </ul>
  );
}

export default StatementList;
