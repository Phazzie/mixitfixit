/**
 * @module localStorageManager
 * @description This module provides functions to interact with the browser's local storage.
 * It offers a way to store, retrieve, and remove data from local storage, with basic error handling.
 */

/**
 * @function getFromLocalStorage
 * @description Retrieves an item from the local storage based on the provided key.
 * @param {string} key - The key of the item to retrieve.
 * @returns {string|null} The value of the item if it exists, or null if the item does not exist or an error occurs.
 * @throws {Error} Throws an error if the key is not a non-empty string.
 * @precondition key must be a non-empty string.
 * @postcondition Returns the value associated with the key, or null.
 */
export const getFromLocalStorage = (key) => {
  if (typeof key !== 'string' || key.trim() === '') {
    throw new Error('Key must be a non-empty string');
  }
  try {
    return localStorage.getItem(key);
  } catch (error) {
    console.error('Error getting item:', error);
    return null;
  }
};

/**
 * @function setInLocalStorage
 * @description Stores an item in the local storage with the provided key and value.
 * @param {string} key - The key to use for storing the item.
 * @param {string} value - The value to store.
 * @throws {Error} Throws an error if the key or value is not a non-empty string.
 * @precondition key must be a non-empty string.
 * @precondition value must be a non-empty string.
 * @postcondition Stores the value with the provided key.
 */
export const setInLocalStorage = (key, value) => {
  if (typeof key !== 'string' || key.trim() === '') {
    throw new Error('Key must be a non-empty string');
  }
  if (typeof value !== 'string' || value.trim() === '') {
    throw new Error('Value must be a non-empty string');
  }
  try {
    localStorage.setItem(key, value);
  } catch (error) {
    console.error('Error setting item:', error);
  }
};

/**
 * @function removeFromLocalStorage
 * @description Removes an item from the local storage based on the provided key.
 * @param {string} key - The key of the item to remove.
 * @throws {Error} Throws an error if the key is not a non-empty string.
 * @precondition key must be a non-empty string.
 * @postcondition The item associated with the key is removed.
 */
export const removeFromLocalStorage = (key) => {
  if (typeof key !== 'string' || key.trim() === '') {
    throw new Error('Key must be a non-empty string');
  }
  try {
    localStorage.removeItem(key);
  } catch (error) {
    console.error('Error removing item:', error);
  }
};
